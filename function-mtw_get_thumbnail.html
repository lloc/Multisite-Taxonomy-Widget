Function mtw_get_thumbnail

Get thumbnail

You can use code like this if you want to override the output of the function:

function my_get_thumbnail( WP_Post $post, array $atts ) {
    if ( !empty( $atts['thumbnail'] ) ) {
        return sprintf(
            '<a href="%s" title="%s">%s</a>',
            $post->mtw_href,
            apply_filters( 'the_title', $post->post_title ),
            $post->mtw_thumb
        );
    }
    return '';
}
add_filter( 'mtw_thumbnail_output_filter', 'my_get_thumbnail' );
